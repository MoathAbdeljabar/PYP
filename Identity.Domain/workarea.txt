[HttpPost("cart/items")]
[ProducesResponseType(typeof(CartItemResponse), StatusCodes.Status201Created)]
[ProducesResponseType(typeof(ErrorResponse), StatusCodes.Status400BadRequest)]
public async Task<IActionResult> AddToCart([FromBody] AddToCartRequest request)
{
    var cartItem = await _cartService.AddItemAsync(request.ProductId, request.Quantity);
    
    // Return 201 with location header pointing to the cart item
    return CreatedAtAction(
        nameof(GetCartItem),  // Name of GET endpoint for cart item
        new { id = cartItem.Id },
        new CartItemResponse
        {
            Id = cartItem.Id,
            ProductId = cartItem.ProductId,
            Quantity = cartItem.Quantity
        });
}
